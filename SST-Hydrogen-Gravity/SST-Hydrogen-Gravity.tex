\section{Cauchy Integral and Circulation Quantization}

        \oint_C \vswirl \cdot d\mathbf{l} =

            0, & \text{if no singularity inside,}\\[4pt]
            2\pi i\, \mathrm{Res}\!\left(\frac{dW}{dz}, z=0\right), & \text{if the axis is enclosed.}

    \end{equation}

    \end{equation}
